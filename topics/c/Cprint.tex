\question You are given the following code below. For each question, put what the C code would print.

\begin{blocksection}

\begin{verbatim}

void foo(int a) {
    a = 61;
}

int main() {
   int y = 20;
   foo(y);
   printf("%d", y);
   return 0;
}


\end{verbatim}

\end{blocksection}
\textbf{Answer}: \rule{1cm}{0.25mm}
