\begin{blocksection}
\question
For each line of RISC-V code, answer what will be the value saved into the registers involved or
the effect on memory.
Assume that x8 contains a valid address in memory, such that Mem[R[x8]] = 0x00000180.

\begin{parts}
\part
lw x9, 0(x8)
What value does x9 now store?

\begin{solution}[0.5in]
0x00000180
\end{solution}

\part
lb x10, 1(x8) 
What value does x10 store? 

\begin{solution}[0.5in]
0x00000001
\end{solution}

\part
lb x11, 0(x8) 
What value does x11 store? 

\begin{solution}[0.5in]
0xFFFFFF80
\end{solution}

\part
sb x11, 3(x8) 
Whatâ€™s the effect on Mem[R[x8]]? 

\begin{solution}[0.5in]
Mem[R[x8]] = 0x80000180
\end{solution}

\part
lbu x12, 0(x8) 
What value does x12 store? 

\begin{solution}[0.5in]
0x00000080
\end{solution}
\end{parts}
