\begin{blocksection}
\question
What is the range of integers represented by a $n$-bit binary number? Your answers should include expressions that use $2^n$.

\begin{parts}
\part
Unsigned:

\begin{solution}[0.5in]
$[0, 2^n - 1]$
\end{solution}

\part
Two's Complement:

\begin{solution}[0.5in]
$[-2^{n - 1}, 2^{n - 1} - 1]$
\end{solution}

\part
Bias (with bias $b$):

\begin{solution}[0.5in]
$[-b, 2^n - 1 - b]$
\end{solution}

\end{parts}

\end{blocksection}
