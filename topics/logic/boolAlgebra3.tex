\question

Simplify the above boolean expression.
\begin{solution}[0.7in]

\begin{split}
A\bar{B}C + A\bar{B}\bar{C} + \bar{A}\bar{B}C &= \bar{B}(AC + A\bar{C} + \bar{A}C) \\
&= \bar{B}(AC + A\bar{C} + AC + \bar{A}C) \\
&= \bar{B}(A(C + \bar{C}) + C(A + \bar{A})) \\
&= \bar{B}(A + C)
\end{split}

\end{solution}
