
\question 
\begin{parts}
\part If we want to tolerate 1 disk failure, which version(s) of RAID should we use?
\begin{checkboxes}
 \choice RAID 0
 \choice RAID 1
 \choice RAID 4
 \choice RAID 5
 \choice None of the above
\end{checkboxes}
\begin{solution}
RAID 1, RAID 4, RAID 5
\end{solution}

\part Which version of RAID is fastest for small random writes?
\begin{checkboxes}
 \choice RAID 0
 \choice RAID 1
 \choice RAID 4
 \choice RAID 5
 \choice None of the above
\end{checkboxes}
\begin{solution}
RAID 0
\end{solution}

\end{parts}

\question
\begin{parts}
\part What is the minimum Hamming distance necessary to allow for Single Error Detection and Single Error Correction?
\begin{solution}[0.5in] 
3
\end{solution}

\part How many bits of data can we cover if we have 7 parity bits?
\begin{solution}[0.5in] 
120
\end{solution}

\part Consider the following codeword we wish to send: 0b10110100. What is the Hamming ECC we should send over to ensure that we can detect and correct a 1-bit error?
\begin{solution}[0.5in] 
0b001001110100
\end{solution}
\end{parts}
